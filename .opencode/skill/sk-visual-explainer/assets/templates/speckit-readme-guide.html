Quick Start

# 1) Clone repository
 git clone https://github.com/MichelKerkmeester/opencode-spec-kit-framework.git
 cd opencode-spec-kit-framework

# 2) Build memory engine
 cd .opencode/skill/system-spec-kit && npm install && npm run build && cd ../../..

 opencode
